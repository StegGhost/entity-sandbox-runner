name: Marketing System

on:
  workflow_dispatch:

jobs:
  run:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Run marketing system
        run: |
          python install/marketing_system.py

      - name: Show outputs
        run: |
          ls -R marketing || true
          ls -R payload/receipts/marketing_system || true

      - name: Commit outputs
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "actions@github.com"
          git add marketing payload/receipts/marketing_system || true
          git commit -m "marketing system update" || echo "no changes"
          git push
