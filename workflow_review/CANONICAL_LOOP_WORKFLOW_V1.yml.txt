name: Canonical Loop

on:
  workflow_dispatch:

jobs:
  canonical_loop:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Run canonical loop
        run: |
          python install/run_canonical_loop.py

      - name: Commit canonical outputs
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "actions@github.com"
          git add docs/canonical payload/feedback payload/receipts/document_compaction payload/receipts/canonical_feedback || true
          git commit -m "canonical loop update" || echo "no changes"
          git push
