name: Document Engine

on:
  workflow_dispatch:
    inputs:
      doc_type:
        description: "Type of document"
        required: true
        type: choice
        options:
          - preprint
          - whitepaper
          - technical_spec
          - governance_model
      topic:
        description: "Topic"
        required: true
        type: string
      output_format:
        description: "Output format"
        required: true
        type: choice
        options:
          - md
          - md+tex
          - full_bundle
      bundle_target:
        description: "Target"
        required: true
        type: choice
        options:
          - ingestion
          - download_only

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Run Document Engine
        run: |
          export PYTHONPATH=$PYTHONPATH:$(pwd)
          python - <<'PY'
          from install.document_pipeline import run
          run(
              doc_type="${{ inputs.doc_type }}",
              topic="${{ inputs.topic }}",
              output_format="${{ inputs.output_format }}",
              target="incoming_bundles" if "${{ inputs.bundle_target }}" == "ingestion" else "."
          )
          PY

      - name: Commit output
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "actions@github.com"
          git add .
          git commit -m "document bundle generated" || echo "no changes"
          git push
