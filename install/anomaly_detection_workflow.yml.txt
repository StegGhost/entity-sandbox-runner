name: Anomaly Detection

on:
  workflow_dispatch:

jobs:
  run:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"
      - run: python install/anomaly_detection.py
      - run: |
          git config user.name "github-actions[bot]"
          git config user.email "actions@github.com"
          git add payload/anomaly_detection payload/receipts/anomaly_detection || true
          git commit -m "anomaly detection update" || echo "no changes"
          git push
